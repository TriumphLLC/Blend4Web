Example
var m_obj = require("objects");
var wb_params =
{
    angle: 45,
    main_frequency: 0.25,
    detail_frequency: 1,
    detail_amplitude: 0.1,
    branch_amplitude: 0.3
};
m_obj.set_wind_bending_params(obj, wb_params);
